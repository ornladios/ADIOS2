  RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

  @ONLY

  OUTPUT ${PROJECT_BINARY_DIR}/$<CONFIG>/bpls.cmake
  INPUT ${PROJECT_BINARY_DIR}/bpls.cmake.gen
)
